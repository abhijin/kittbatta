\renewcommand{\diamond}[4]{\draw[draw=#4,fill=#3,shift={(#1,#2)}] (0,0) --
(.5,.5) -- (1,0) -- (.5,-.5) -- cycle}
\newcommand{\downtriangle}[4]{\draw[draw=#4,fill=#3,shift={(#1,#2)}] (0,0)
-- (1,0) -- (.5,-.5)}
\newcommand{\righttriangle}[4]{\draw[draw=#4,fill=#3,shift={(#1,#2)}] (0,0)
-- (0,1) -- (.5,.5)}
\newcommand{\uptriangle}[4]{\draw[draw=#4,fill=#3,shift={(#1,#2)}] (0,0)
-- (1,0) -- (.5,.5)}


\newcommand{\diamondgrid}[7]{
\begin{scope}[shift={(#1,#2)}]
    \fill[#7,shift={(0,-.5)}] (0,0) rectangle (#3+1,#4+1);
\foreach \x in {0,...,#3}
    \foreach \y in {0,...,#4}
        \diamond{\x}{\y}{#5}{#6};
\end{scope}
}

\newcommand{\fourdiamonds}[4]{
\begin{scope}[shift={(#1,#2)}]
\diamond{0}{0}{#3}{#4};
\diamond{1}{0}{#3}{#4};
\diamond{.5}{.5}{#3}{#4};
\diamond{.5}{-.5}{#3}{#4};
\end{scope}
}

\newcommand{\eightdiamonds}[4]{
\begin{scope}[shift={(#1,#2)}]
\diamond{0}{0}{#3}{#4};
\diamond{.5}{.5}{#3}{#4};
\diamond{1}{1}{#3}{#4};
\diamond{1.5}{.5}{#3}{#4};

\diamond{1.5}{-.5}{#3}{#4};
\diamond{1}{-1}{#3}{#4};
\diamond{.5}{-.5}{#3}{#4};
\diamond{1}{0}{#3}{#4};
\end{scope}
}
